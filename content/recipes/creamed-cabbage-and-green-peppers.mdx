ingredients:
  - "2 tablespoons minced onion"
  - "1 minced green pepper"
  - "1 1/2 tablespoons butter"
  - "2 tablespoons flour"
  - "1 cup milk"
  - "1 pint cooked shredded cabbage"
  - "Salt, pepper"
