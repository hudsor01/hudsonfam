servings: ""

cookTime: ""
ingredients:
  - "3 cups brown sugar"
  - "1/2 cup water"
  - "1/2 cup vinegar"
  - "1/2 ounce ginger root"
  - "1/2 tablespoon whole cloves"
  - "1/2 tablespoon stick cinnamon"
  - "1 quart yellow tomatoes"
instructions:
  - "Boil together sugar, water, vinegar, and spices for five minutes. Pierce tomatoes with fork and boil in syrup fifteen minutes, and let stand overnight. Remove tomatoes from the syrup, and boil it five minutes or until thick; then heat tomatoes in syrup and seal in hot sterilized jars."
