servings: "8"
prepTime: ""
cookTime: "4 hours"
ingredients:
  - "3 cups milk"
  - "1/2 cup corn meal"
  - "1 teaspoon cinnamon"
  - "1 teaspoon salt"
  - "1 1/2 cups brown sugar"
  - "1 pint milk (cold)"
  - "2 tablespoons butter"
  - "1 quart sweet apples"
instructions:
  - "Scald three cups milk, sift in corn meal, stirring rapidly and cook five minutes. Remove from fire, add cinnamon, salt, sugar, cold milk, butter, and apples cut in eighths. Bake in a deep covered dish in a moderate oven. Bake in the oven for 4 hours at 300 degrees. Servings, 8."
