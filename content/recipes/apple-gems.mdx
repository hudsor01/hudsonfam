servings: "12"

cookTime: "25 minutes"
ingredients:
  - "1 cup cornmeal"
  - "1 1/2 cups flour"
  - "1/2 teaspoon salt"
  - "2 tablespoons sugar"
  - "5 teaspoons baking powder"
  - "2 tablespoons molasses"
  - "1 cup milk"
  - "4 sour apples"
instructions:
  - "Sift dry ingredients, add molasses and milk and beat until smooth. Peel and chop apples, adding them to batter. Put into muffin-tins and bake in a moderately hot oven for 25 minutes at 375 degrees."
