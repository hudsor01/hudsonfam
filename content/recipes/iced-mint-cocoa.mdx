category: "Beverages"

servings: "6"
prepTime: ""
cookTime: ""
ingredients:
  - "6 tablespoons cocoa"
  - "1 cup boiling water"
  - "4 sprigs of mint"
  - "3/4 cup sugar"
  - "1 quart milk"
  - "1/2 teaspoon vanilla"
instructions:
  - "Mix cocoa with boiling water. Add mint and cook until thick. Add sugar and milk. Bring to boiling point and remove from fire and strain. Cool, add vanilla. Ice and serve. Charged water may be added. Servings, 6."
