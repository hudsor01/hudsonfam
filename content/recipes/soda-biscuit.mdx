category: "Bread"
status: "published"

servings: ""
prepTime: ""
cookTime: "15 minutes"
ingredients:
  - "1 pint flour"
  - "1 teaspoon salt"
  - "1/2 teaspoon soda"
  - "1 teaspoon baking powder"
  - "1 tablespoon lard"
  - "1 cup buttermilk"
instructions:
  - "Sift flour, salt, soda, and baking powder together."
  - "Mix in lard thoroughly, then add buttermilk."
  - "Roll out on floured board, cut with biscuit-cutter and bake in the oven for 15 minutes at 400 degrees. Handle as little as possible."
