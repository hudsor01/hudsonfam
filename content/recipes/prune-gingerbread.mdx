servings: "8"
prepTime: ""
cookTime: "30 minutes"
ingredients:
  - "2 cups flour"
  - "2 teaspoons soda"
  - "1/2 teaspoon salt"
  - "2 teaspoons ginger"
  - "1 teaspoon cinnamon"
  - "1 cup molasses"
  - "1 cup sour milk"
  - "1 cup chopped prunes"
  - "4 tablespoons melted drippings"
instructions:
  - "Mix and sift dry ingredients, add molasses, sour milk, prunes and drippings, beat well; pour into shallow pan. Bake in a moderate oven in the oven for 30 minutes at 350 degrees."
