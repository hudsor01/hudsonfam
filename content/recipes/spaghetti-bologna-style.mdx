servings: "12"

cookTime: ""
ingredients:
  - "1/4 pound veal, ground"
  - "1/4 pound pork, ground"
  - "1/4 pound beef, ground"
  - "1/4 pound sausage"
  - "1 tablespoon butter"
  - "1 cup onion, chopped"
  - "1 clove garlic, chopped"
  - "1 green pepper, chopped"
  - "1 sprig parsley, cut"
  - "4 ounces mushrooms, chopped"
  - "1/2 cup tomato puree"
  - "1/4 cup tomato catsup"
  - "1/2 cup water"
  - "1/4 teaspoon Worcestershire sauce"
  - "1 teaspoon lemon juice"
  - "1 teaspoon salt"
  - "Pinch pepper"
  - "1/2 pound spaghetti"
  - "Grated cheese"
instructions:
  - "Brown meats in butter in iron skillet, add onion, garlic, peppers and parsley. Cook until slightly brown and add mushrooms, puree, catsup, one-half cup water, Worcestershire sauce, lemon juice, and salt and pepper. Simmer one and one-half hours. Cook spaghetti in salted water until soft, and cook sauce and spaghetti together ten minutes. Serve on platter and sprinkle with grated cheese."
