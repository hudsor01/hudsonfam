servings: "8"
prepTime: ""
cookTime: ""
ingredients:
  - "1 1/2 cups coffee"
  - "3 pints boiling water"
  - "1/2 pint cream"
  - "1/3 cup sugar"
  - "1/2 teaspoon vanilla"
instructions:
  - "Put coffee in cheese-cloth bag, add water and boil three minutes. Let stand ten minutes then remove bag and keep coffee hot until serving-time. Beat cream until stiff, add sugar and vanilla and serve a spoonful on each cup of coffee."
