servings: "8"

cookTime: "1 hour"
ingredients:
  - "Pinch salt"
  - "1 cup egg whites"
  - "1 1/4 cups fine granulated sugar"
  - "1 cup sifted flour"
  - "1 teaspoon cream of tartar"
  - "3/4 teaspoon almond extract"
  - "1/2 teaspoon vanilla"
instructions:
  - "Add salt to egg whites and beat until stiff and dry. Fold in sugar, then flour sifted several times with cream of tartar. Add flavorings and bake in a moderate oven for 1 hour at 325 degrees."
