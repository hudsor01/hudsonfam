servings: "6"
prepTime: ""
cookTime: ""
ingredients:
  - "1 tablespoon butter"
  - "1 tablespoon flour"
  - "1 1/4 cups thin cream"
  - "1 1/2 tablespoons chopped green pepper"
  - "1 1/2 tablespoons chopped pimiento"
  - "1 tablespoon grated onion"
  - "2 cups tuna fish"
  - "1 egg"
instructions:
  - "Melt the butter, add the flour, and add the cream gradually, bringing to the boiling point. Add the green pepper, pimiento, onion, and tuna fish."
  - "Add the egg, slightly beaten, and cook three minutes. Serve in timbale cases, patty cases, or on toast."
