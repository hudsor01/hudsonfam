title: "Ginger Drop Cookies"
category: "Cookies"
status: "published"

servings: "36"
prepTime: ""
cookTime: "15 minutes"
ingredients:
  - "1/2 cup brown sugar"
  - "1 tablespoon melted butter"
  - "1/2 cup molasses"
  - "1 egg"
  - "1/2 cup sour cream"
  - "1 cup raisins"
  - "2 1/2 cups flour"
  - "1/2 teaspoon salt"
  - "1 teaspoon soda"
  - "1/2 teaspoon cream of tartar"
  - "1/2 teaspoon clove"
  - "1 teaspoon ginger"
  - "1 teaspoon cinnamon"
instructions:
  - "Mix sugar and melted butter, add molasses, beaten egg, sour cream, raisins and the flour mixed and sifted with dry ingredients. Drop from teaspoon on buttered tins. Bake in a moderate oven for 15 minutes at 350 degrees."
