servings: "14"
prepTime: ""
cookTime: "40 minutes"
ingredients:
  - "2/3 cup graham flour"
  - "1 cup entire wheat flour"
  - "1/2 cup flour"
  - "1/4 cup sugar"
  - "1 teaspoon salt"
  - "2 teaspoons baking powder"
  - "1 teaspoon soda"
  - "3/4 cup fruit (dates, nuts, raisins)"
  - "3 tablespoons molasses"
  - "1 cup sour milk"
  - "1 teaspoon mapleine"
  - "2 tablespoons melted shortening"
instructions:
  - "Mix and sift dry ingredients. Add fruit cut rather fine, molasses, milk, and mapleine. Beat well, add melted shortening and bake in muffin tins in a moderately hot oven for 40 minutes at 375 degrees."
