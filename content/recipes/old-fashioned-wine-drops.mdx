category: "Cookies"
status: "published"
order: 9201
servings: "48"
prepTime: ""
cookTime: "15 minutes"
ingredients:
  - "3/4 cup lard or other shortening"
  - "1/2 cup brown sugar"
  - "1 egg"
  - "1/2 cup molasses"
  - "1/2 cup milk"
  - "2 1/2 cups flour"
  - "1 teaspoon soda"
  - "1 teaspoon salt"
  - "1/2 teaspoon clove"
  - "1 teaspoon cinnamon"
  - "1 teaspoon nutmeg"
  - "7/8 cup currants"
instructions:
  - "Cream shortening and sugar together, add egg, molasses, milk, and flour mixed and sifted with other dry ingredients. Mix well and add currants floured. Drop on greased tins about two inches apart. Bake in a moderate oven for 15 minutes at 350 degrees."
