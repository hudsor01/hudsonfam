servings: "9"
prepTime: ""
cookTime: ""
ingredients:
  - "1 quart crushed pineapple"
  - "1 cup sugar"
  - "1 tablespoon gelatine"
  - "1/2 cup cold water"
  - "2 tablespoons lemon juice"
  - "1 cup orange juice"
  - "1 cup cream, whipped"
instructions:
  - "Boil the pineapple and sugar together for fifteen minutes. Add gelatine which has been softened in the cold water. Add lemon and orange juices. Freeze, adding cream when partly frozen. Servings, 9."
