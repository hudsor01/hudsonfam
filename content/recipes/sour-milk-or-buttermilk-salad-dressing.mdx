servings: ""
prepTime: ""
cookTime: ""
ingredients:
  - "1/4 cup butter"
  - "1/4 cup sugar"
  - "4 tablespoons flour"
  - "2 teaspoons salt"
  - "1/2 teaspoon mustard"
  - "2 eggs"
  - "1 1/2 cups sour milk or buttermilk"
  - "3/4 cup weak vinegar"
instructions:
  - "Melt the butter, add the sugar, flour, and seasonings. Add beaten eggs and the milk and stir over hot water until thick, then add the vinegar, slowly, stirring constantly until it thickens again. Remove from the heat, chill and serve."
