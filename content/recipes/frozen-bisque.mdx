title: "Frozen Bisque"
category: "Dessert"
status: "published"
order: 13702
servings: "8"
prepTime: ""
cookTime: ""
ingredients:
  - "1 pint cream"
  - "2 eggs"
  - "3/4 cup powdered sugar"
  - "6 oatmeal or graham crackers"
  - "1/2 cup cocoanut"
  - "1 1/2 teaspoons vanilla"
instructions:
  - "Whip cream, add egg yolks well beaten, sugar, crackers rolled fine, cocoanut, and vanilla. Fold in stiffly beaten egg whites. Put in freezer or mold, pack in ice and salt, and let stand three hours."
