const N8N_WEBHOOK_BASE =

async function fireWebhook(
  path: string,
  body: Record<string, unknown>
): Promise<void> {
  try {
    await fetch(`${N8N_WEBHOOK_BASE}/webhook/${path}`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(body),
    });
  } catch {

  }
}
