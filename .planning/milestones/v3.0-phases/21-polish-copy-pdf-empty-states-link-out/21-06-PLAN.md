    ```typescript
    import type { CompanyResearch } from "@/lib/jobs-db";

    /**
     * Returns `true` when a company_research row exists but every LLM-derived
     * field is null / empty / zero-length. Drives the AI-RENDER-04
     * "generated but empty" branch for Company Intel.
     *
     * Research Note (21-RESEARCH.md §Open Question in UI-SPEC §3):
     *   - 0 company_research rows exist in the DB today (Phase 23 will
     *     produce the first ones).
     *   - n8n's company-research workflow may emit "Could not find info"
     *     as `ai_summary` rather than null — in which case the strict
     *     all-null predicate below becomes dead code.
     *   - Planner (Phase 21) refines after Phase 23 ships real rows.
     *
     * Conservative today: row is "empty" if NONE of the LLM-derived columns
     * are populated. A single populated field (even a bad one) means the
     * rich rendering path fires, giving the owner the most information.
     */
    export function isCompanyResearchEmpty(cr: CompanyResearch): boolean {
      return (
        !cr.ai_summary?.trim() &&
        (!cr.tech_stack || cr.tech_stack.length === 0) &&
        !cr.recent_news?.trim() &&
        cr.glassdoor_rating === null &&
        cr.employee_count === null &&
        cr.funding_stage === null &&
        cr.salary_range_min === null &&
        cr.salary_range_max === null
      );
    }
    ```
