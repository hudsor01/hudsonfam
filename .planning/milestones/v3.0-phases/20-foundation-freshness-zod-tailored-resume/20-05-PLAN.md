### File 1 — `src/app/(admin)/admin/jobs/tailored-resume-section.tsx`

Create the component, matching UI-SPEC §1 render tree exactly. Use the props shape below so Plan 06 can pass pre-computed freshness from the server.

```typescript
"use client";

import { FileText } from "lucide-react";
import { Streamdown } from "streamdown";
import { FreshnessBadge } from "./freshness-badge";

export interface ResumeFreshness {
  relativeTime: string; // e.g. "3 days ago"
  isStale: boolean;
  ageDays: number;
}

export interface TailoredResumeView {
  content: string;
  model_used: string | null;
  freshness: ResumeFreshness;
}

interface Props {
  resume: TailoredResumeView | null;
}

/**
 * Renders the tailored resume markdown block inside the job detail sheet.
 *
 * UI-SPEC §1:
 *   - Heading: "Tailored Resume" + FileText icon (size-4), matches Cover Letter cadence
 *   - Meta row: FreshnessBadge (Generated {relativeTime} · {model_used}), right-aligned
 *   - Body: Streamdown-rendered markdown inside bg-card/50 rounded-lg p-4 border border-border max-h-96
 *
 * XSS posture (AI-SAFETY-01 + RESEARCH.md Q2):
 *   - Streamdown is invoked with `skipHtml` — raw HTML in LLM output is NEVER parsed.
 *   - Defense in depth: Plan 07's CSP header on /admin/* blocks inline scripts via
 *     nonce + object-src 'none' + frame-ancestors 'none'.
 *   - Streamdown's default pipeline includes rehype-raw; `skipHtml` fully bypasses
 *     it, which is the recommended primary choice per RESEARCH.md.
 *
 * linkSafety is disabled: the admin surface is trusted single-owner; the modal
 * confirmation adds friction. Re-enable if a family-member-viewable markdown
 * surface is ever added.
 */
export function TailoredResumeSection({ resume }: Props) {
  if (!resume) return null;

  return (
    <div className="space-y-3">
      <div className="flex items-center justify-between">
        <h3 className="text-sm font-semibold flex items-center gap-1.5">
          <FileText className="size-4" />
          Tailored Resume
        </h3>
        <FreshnessBadge
          relativeTime={resume.freshness.relativeTime}
          modelUsed={resume.model_used}
          isStale={resume.freshness.isStale}
          ageDays={resume.freshness.ageDays}
        />
      </div>
      <div className="text-sm text-muted-foreground bg-card/50 rounded-lg p-4 border border-border max-h-96 overflow-y-auto">
        <Streamdown skipHtml linkSafety={{ enabled: false }}>
          {resume.content}
        </Streamdown>
      </div>
    </div>
  );
}
```

### File 2 — `src/__tests__/components/tailored-resume-xss.test.tsx`

XSS regression guard. Ships in the same PR as the Streamdown integration per CONTEXT.md D-13.

```typescript
import { describe, it, expect } from "vitest";
import { render } from "@testing-library/react";
import { Streamdown } from "streamdown";

describe("Streamdown XSS protection with skipHtml", () => {
  const XSS_PAYLOADS = [
    "<script>alert(1)</script>",
    '<iframe src="javascript:alert(1)"></iframe>',
    "<img src=x onerror=alert(1)>",
  ];

  for (const payload of XSS_PAYLOADS) {
    it(`renders '${payload}' as literal text, not as executable DOM`, () => {
      const { container } = render(
        <Streamdown skipHtml linkSafety={{ enabled: false }}>
          {payload}
        </Streamdown>
      );

      // Assertion 1: no <script> element in the rendered output
      expect(container.querySelector("script")).toBeNull();

      // Assertion 2: no <iframe> element
      expect(container.querySelector("iframe")).toBeNull();

      // Assertion 3: no <img> element with an onerror attribute
      const imgs = container.querySelectorAll("img");
      imgs.forEach((img) => {
        expect(img.getAttribute("onerror")).toBeNull();
      });

      // Assertion 4: the raw payload text is visible (rendered as text, not stripped)
      expect(container.textContent).toContain(payload);
    });
  }

  it("renders safe markdown (heading, bold, list) normally", () => {
    const md = "# Title\n\n**bold**\n\n- item 1\n- item 2";
    const { container } = render(
      <Streamdown skipHtml linkSafety={{ enabled: false }}>
        {md}
      </Streamdown>
    );
    expect(container.querySelector("h1")).not.toBeNull();
    expect(container.querySelector("strong")).not.toBeNull();
    const lis = container.querySelectorAll("li");
    expect(lis.length).toBeGreaterThanOrEqual(2);
  });

  it("does not render raw HTML tags present in markdown even if they look harmless", () => {
    const { container } = render(
      <Streamdown skipHtml linkSafety={{ enabled: false }}>
        {"<b>raw bold</b> and <u>raw underline</u>"}
      </Streamdown>
    );
    // skipHtml → these tags appear as text, not as elements
    expect(container.querySelector("b")).toBeNull();
    expect(container.querySelector("u")).toBeNull();
    expect(container.textContent).toContain("<b>raw bold</b>");
    expect(container.textContent).toContain("<u>raw underline</u>");
  });
});
```

import { describe, it, expect } from "vitest";
import { render } from "@testing-library/react";
import {
  TailoredResumeSection,
  type TailoredResumeView,
} from "@/app/(admin)/admin/jobs/tailored-resume-section";

const freshView: TailoredResumeView = {
  content: "# Richard Hudson\n\n**Senior Engineer**\n\n- item",
  model_used: "gpt-4o-mini",
  freshness: { relativeTime: "3 hours ago", isStale: false, ageDays: 0 },
};

const staleView: TailoredResumeView = {
  content: "# Resume",
  model_used: "gpt-4o-mini",
  freshness: { relativeTime: "20 days ago", isStale: true, ageDays: 20 },
};

describe("TailoredResumeSection", () => {
  it("renders heading 'Tailored Resume' with FileText icon", () => {
    const { container } = render(<TailoredResumeSection resume={freshView} />);
    const heading = container.querySelector("h3");
    expect(heading).not.toBeNull();
    expect(heading?.textContent).toContain("Tailored Resume");
    // lucide FileText renders an <svg>
    expect(heading?.querySelector("svg")).not.toBeNull();
  });

  it("renders markdown content via Streamdown (not plaintext)", () => {
    const { container } = render(<TailoredResumeSection resume={freshView} />);
    // Streamdown produces an <h1> for "# Richard Hudson" and <strong> for **bold**
    expect(container.querySelector("h1")).not.toBeNull();
    expect(container.querySelector("strong")).not.toBeNull();
    // Container body uses UI-SPEC styling (bg-card/50, max-h-96)
    const body = container.querySelector(".max-h-96");
    expect(body).not.toBeNull();
    expect(body?.className).toContain("bg-card/50");
    expect(body?.className).toContain("overflow-y-auto");
  });

  it("returns null when resume is null (Phase 20 hides section; empty state is Phase 21)", () => {
    const { container } = render(<TailoredResumeSection resume={null} />);
    expect(container.firstChild).toBeNull();
  });

  it("renders the amber stale dot when freshness.isStale is true", () => {
    const { container } = render(<TailoredResumeSection resume={staleView} />);
    const dot = container.querySelector(".bg-warning");
    expect(dot).not.toBeNull();
    expect(dot?.getAttribute("aria-label")).toBe("Stale artifact");
  });

  it("renders model_used next to the relative time when present", () => {
    const { container } = render(<TailoredResumeSection resume={freshView} />);
    expect(container.textContent).toContain("Generated 3 hours ago");
    expect(container.textContent).toContain("gpt-4o-mini");
  });

  it("does NOT use whitespace-pre-wrap (that's the old cover-letter plaintext style)", () => {
    const { container } = render(<TailoredResumeSection resume={freshView} />);

    expect(body?.className || "").not.toContain("whitespace-pre-wrap");
  });
});

### Gotchas

- CLAUDE.md: NO hardcoded Tailwind colors — use `text-muted-foreground`, `bg-card/50`, `border-border`.
- Do NOT bare-render `<Streamdown>{content}</Streamdown>` — you MUST pass `skipHtml`. RESEARCH.md Q2 Pitfall 2 explains why: Streamdown's DEFAULT includes rehype-raw. Without `skipHtml`, the XSS tests will fail because the payload gets sanitized-away instead of literalized.
- Do NOT import from `streamdown/styles.css` — only needed when the `animated` prop is in use (RESEARCH.md §Q2).
- Do NOT use `linkSafety={{ enabled: true }}` or omit the prop — omission defaults to on. RESEARCH.md §Open Questions #3 recommends `{{ enabled: false }}` for the owner-only admin surface.
- Add `"use client"` as the first line — `job-detail-sheet.tsx` is already a client component but this file is tested standalone, so it must declare its own directive.
